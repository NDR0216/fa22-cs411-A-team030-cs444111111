Popular Channels in the 2020s

SELECT channel_title, COUNT(video_id) AS cnt
FROM workout.Workout_Video_Trainer NATURAL JOIN workout.Channels
WHERE video_viewCount > 1000000 AND publish_date LIKE "202%"
GROUP BY channel_id
ORDER BY cnt DESC
LIMIT 15;
